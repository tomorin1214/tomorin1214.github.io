<!-- 
  ===== 嵌入网易云音乐的方法 =====
  
  从网易云音乐网页版打开歌曲/歌单页面，复制网址里的 ID：
  歌曲示例：https://music.163.com/#/song?id=1863312160 → ID 就是 1863312160
  
  然后把下面的 "你的歌曲ID" 替换成实际的 ID 即可
-->

<!-- 单曲 -->
<meting-js server="netease" type="song" id="你的歌曲ID" theme="#C20C0C"></meting-js>

<!-- 其他类型参考：
  歌单：<meting-js server="netease" type="playlist" id="你的歌单ID"></meting-js>
  专辑：<meting-js server="netease" type="album" id="你的专辑ID"></meting-js>
  多首：id="ID1,ID2,ID3"
-->
